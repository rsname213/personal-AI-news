**fetchers/gwern.py** — HTML scraper of gwern.net/blog/index:

Gwern has no native RSS (confirmed: GitHub issue #11 on gwern/gwern.net, open since 2015).

```python
"""
Gwern Branwen fetcher.
Scrapes gwern.net/blog/index — no native RSS exists.
"""
import hashlib
import httpx
from bs4 import BeautifulSoup
from datetime import datetime, timezone, timedelta
from models import RawArticle

GWERN_BLOG_INDEX = "https://gwern.net/blog/index"

def fetch() -> list[RawArticle]:
    cutoff = datetime.now(timezone.utc) - timedelta(hours=25)
    try:
        response = httpx.get(GWERN_BLOG_INDEX, timeout=15, follow_redirects=True,
                             headers={"User-Agent": "Mozilla/5.0 (compatible; newsletter-bot/1.0)"})
        response.raise_for_status()
    except Exception as e:
        print(f"[WARN] Gwern: fetch failed: {e}")
        return []

    soup = BeautifulSoup(response.text, "lxml")
    articles = []

    # gwern.net/blog/index lists posts with dates
    # Inspect actual structure: look for <article>, <section>, or <ul> with date+link pairs
    # Common patterns on gwern.net: <p> elements with <a> links and date metadata
    # The exact selectors MUST be verified against the live page during implementation.
    # Strategy: find all links inside the main content area, look for sibling/parent date text

    # Attempt to parse common gwern blog index patterns:
    for link in soup.select("article a[href], .post-list a[href], #main a[href], section a[href]"):
        href = link.get("href", "")
        if not href or href.startswith("#") or href.startswith("http") and "gwern.net" not in href:
            continue
        # Normalize relative URLs
        if href.startswith("/"):
            href = f"https://gwern.net{href}"
        title = link.get_text(strip=True)
        if not title:
            continue

        # Look for date in parent/sibling elements
        parent = link.parent
        date_text = ""
        for elem in [parent, parent.parent if parent else None]:
            if elem:
                text = elem.get_text()
                # Look for year pattern like 2024, 2025, 2026
                import re
                date_match = re.search(r'\b(202[3-9]|2030)\b', text)
                if date_match:
                    date_text = date_match.group(0)
                    break

        # If no date found, skip — cannot determine recency
        if not date_text:
            continue

        # Parse year-only date as Jan 1 of that year (conservative — will miss most recent)
        # For better precision: fetch the individual page to get exact date
        # For Phase 1: year-level filtering is sufficient; Gwern publishes ~10 posts/year
        try:
            published_at = datetime(int(date_text), 1, 1, tzinfo=timezone.utc)
        except ValueError:
            continue

        # Only include articles from current year to avoid old posts
        current_year = datetime.now(timezone.utc).year
        if published_at.year < current_year:
            continue

        articles.append(RawArticle(
            id=hashlib.md5(href.encode()).hexdigest(),
            url=href,
            title=title,
            content_text="",  # Not fetched at scrape time — too slow for Phase 1
            published_at=published_at,
            source_name="Gwern Branwen",
            source_category="Personal Blogs",
        ))

    print(f"[OK] Gwern Branwen: {len(articles)} article(s)")
    if len(articles) == 0:
        print("[WARN] Gwern Branwen: returned 0 articles — HTML structure may have changed")
    return articles
```

IMPORTANT: The gwern.net HTML structure must be inspected during implementation. The selector logic above (`article a[href]`, `.post-list a[href]`, etc.) is a best-effort skeleton — the actual selectors must match what exists on the live page. Fetch the page, inspect the BeautifulSoup tree, and update selectors before marking this task done.

**fetchers/anthropic_blog.py** — Community RSS with scraper fallback:

```python
"""
Anthropic Blog fetcher.
Primary: Community-maintained RSS feed (taobojlen/anthropic-rss-feed).
Fallback: Scrape anthropic.com/news if community feed returns 0 articles.
"""
import hashlib
import httpx
from bs4 import BeautifulSoup
from datetime import datetime, timezone, timedelta
from fetchers.rss import fetch_feed
from models import RawArticle

ANTHROPIC_COMMUNITY_FEED = "https://raw.githubusercontent.com/taobojlen/anthropic-rss-feed/main/anthropic_news_rss.xml"
ANTHROPIC_NEWS_URL = "https://www.anthropic.com/news"

def fetch() -> list[RawArticle]:
    # Try community feed first
    articles = fetch_feed(ANTHROPIC_COMMUNITY_FEED, "Anthropic", "Anthropic")
    if articles:
        return articles

    # Fallback: scrape anthropic.com/news
    print("[WARN] Anthropic: community feed returned 0 articles — trying direct scrape")
    return _scrape_anthropic_news()

def _scrape_anthropic_news() -> list[RawArticle]:
    cutoff = datetime.now(timezone.utc) - timedelta(hours=25)
    try:
        response = httpx.get(ANTHROPIC_NEWS_URL, timeout=15, follow_redirects=True,
                             headers={"User-Agent": "Mozilla/5.0 (compatible; newsletter-bot/1.0)"})
        response.raise_for_status()
    except Exception as e:
        print(f"[WARN] Anthropic scrape fallback failed: {e}")
        return []

    soup = BeautifulSoup(response.text, "lxml")
    articles = []

    # anthropic.com/news uses article cards with title, date, and href
    # Inspect actual structure during implementation — adjust selectors as needed
    for card in soup.select("a[href*='/news/']"):
        href = card.get("href", "")
        if not href:
            continue
        if href.startswith("/"):
            href = f"https://www.anthropic.com{href}"
        title_el = card.select_one("h2, h3, [class*='title']")
        title = title_el.get_text(strip=True) if title_el else card.get_text(strip=True)
        if not title:
            continue

        # Use current time as fallback date — we can't determine recency without page date
        # This means all scraped Anthropic articles will pass the 25h filter; rely on cap
        articles.append(RawArticle(
            id=hashlib.md5(href.encode()).hexdigest(),
            url=href,
            title=title,
            content_text="",
            published_at=datetime.now(timezone.utc),  # Fallback — date unknown from card
            source_name="Anthropic",
            source_category="Anthropic",
        ))

    # Deduplicate by URL
    seen = set()
    unique = []
    for a in articles:
        if a.url not in seen:
            seen.add(a.url)
            unique.append(a)

    print(f"[OK] Anthropic (scraped fallback): {len(unique)} article(s)")
    return unique[:5]  # Cap at 5 for fallback path
```
